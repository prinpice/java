# Test03

```java

public class Test03 {
	// 증감연산자 : ++, --
	public static void main(String[] args) {
		int num = 10;
		++num; // num = num + 1 (num에 1 증가)
		System.out.println(num);
		
		num++; // num = num + 1 (num에 1 증가)
		System.out.println(num);
		
		--num; // num = num - 1 (num에 1 감소)
		System.out.println(num);
		
		num--; // num = num - 1 (num에 1 감소)
		System.out.println(num);

	}
}

```

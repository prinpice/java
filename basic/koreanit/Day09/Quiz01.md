# Quiz01

```
// 자동줄정렬 : ctrl + shift +f
// 한 줄 삭 제 : 드래그 후 ctrl + d
// 줄 옮기기 : alt + 화살표 위/아래
// 자동 주석 : ctrl + shift + c
// 자동완성 : ctrl + space
public class Quiz01 {
	public static void main(String[] args) {
		// 구구단 4단 출력
		
		int num = 1;
		
		while(num < 10) {
			System.out.println(4 +"*"+ num+ "="+ 4*num);
			num++;
		}
	}
}
```
